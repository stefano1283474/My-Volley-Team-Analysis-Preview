Public Sub ImportaScout()
    Dim f As FileDialog
    Dim filePath As String
    Dim srcWB As Workbook
    Dim dstWB As Workbook
    Dim workWB As Workbook ' <-- Nuovo workbook per il file di lavoro
    Dim srcWS As Worksheet
    Dim dstWS As Worksheet
    Dim rng As Range
    Dim arr As Variant
    Dim r As Long, c As Long
    Dim dstCell As Range
    Dim wasCalc As XlCalculation
    Dim wasScreen As Boolean
    Dim wasEvents As Boolean
    Dim wasAlerts As Boolean
    Dim basePath As String
    Dim absCol As Long
    
    Dim vItem As Variant
    Dim folderPath As String
    Dim baseFileName As String
    Dim newFilePath As String
    Dim slashPos As Long
    Dim extPos As Long

    ' Il file master (Scout Default) rimane pulito
    Set dstWB = ThisWorkbook

    Set f = Application.FileDialog(msoFileDialogFilePicker)
    With f
        .Title = "Seleziona file esportato/i dall'app"
        .AllowMultiSelect = True
        .Filters.Clear
        .Filters.Add "Excel", "*.xlsx;*.xlsm;*.xls"
        basePath = dstWB.Path & "\Scout\"
        If Len(Dir(basePath, vbDirectory)) > 0 Then
            .InitialFileName = basePath
        End If
        If .Show <> -1 Then Exit Sub
    End With

    wasCalc = Application.Calculation
    wasScreen = Application.ScreenUpdating
    wasEvents = Application.EnableEvents
    wasAlerts = Application.DisplayAlerts

    ' Disattiva aggiornamenti a schermo per rendere il processo invisibile e velocissimo
    Application.ScreenUpdating = False
    Application.EnableEvents = False
    Application.DisplayAlerts = False
    Application.Calculation = xlCalculationManual

    For Each vItem In f.SelectedItems
        filePath = CStr(vItem)

        If LCase(filePath) <> LCase(dstWB.FullName) Then
            
            ' 1. Estrae percorso e nome per il nuovo file
            slashPos = InStrRev(filePath, "\")
            extPos = InStrRev(filePath, ".")
            
            folderPath = Left$(filePath, slashPos)
            If extPos > slashPos Then
                baseFileName = Mid$(filePath, slashPos + 1, extPos - slashPos - 1)
            Else
                baseFileName = Mid$(filePath, slashPos + 1)
            End If

            newFilePath = folderPath & baseFileName & ".xlsm"

            ' 2. Crea una copia "Vergine" del file Master
            dstWB.SaveCopyAs newFilePath
            
            ' 3. Apre la copia per scriverci dentro i dati (senza toccare il master)
            Set workWB = Workbooks.Open(newFilePath)

            ' 4. Apre il file di origine con i dati dell'app
            Set srcWB = Workbooks.Open(filePath, ReadOnly:=True)

            ' 5. Importazione
            For Each srcWS In srcWB.Worksheets
                Set dstWS = Nothing
                On Error Resume Next
                Set dstWS = workWB.Worksheets(srcWS.Name)
                On Error GoTo 0

                If Not dstWS Is Nothing Then
                    Set rng = srcWS.UsedRange
                    If Not rng Is Nothing Then
                        arr = rng.Value
                        If IsArray(arr) Then
                            For r = 1 To UBound(arr, 1)
                                For c = 1 To UBound(arr, 2)
                                    absCol = rng.Column + c - 1
                                    Set dstCell = dstWS.Cells(rng.Row + r - 1, absCol)
                                    If Not dstCell.HasFormula Then
                                        If dstWS.Name = "El. Gioc." And absCol = 2 Then
                                            dstCell.Value = FormatPlayerNumber(arr(r, c))
                                        Else
                                            dstCell.Value = arr(r, c)
                                        End If
                                    End If
                                Next c
                            Next r
                        End If
                    End If
                End If
            Next srcWS

            ' 6. Chiude e salva solo il file copia compilato. Il master resta immutato!
            srcWB.Close SaveChanges:=False
            workWB.Close SaveChanges:=True

        End If
    Next vItem

    ' Ripristino configurazioni iniziali Excel
    Application.Calculation = wasCalc
    Application.DisplayAlerts = wasAlerts
    Application.EnableEvents = wasEvents
    Application.ScreenUpdating = wasScreen
    
    MsgBox "Elaborazione di " & f.SelectedItems.Count & " file completata con successo!" & vbCrLf & "I nuovi file .xlsm sono stati generati nella cartella di origine.", vbInformation, "Importazione Multipla Terminata"

End Sub
